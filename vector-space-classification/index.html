...and plugins!

mosho.plugin({
  name: "scaleActiveSlide",
  preShow: function (evt) {
    prev = evt.detail.prevSlide
    next = evt.detail.nextSlide
    if (prev) { prev.scale(0.25) }
    if (next) { next.scale(4) }
  }
})
